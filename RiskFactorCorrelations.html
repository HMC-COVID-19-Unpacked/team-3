1. Cardiovascular mortality and Covid-19 Number of Cases

#load data
library(readxl)
covidData <- read_excel("~/Desktop/0423/Age&Cardio/mapping_cardiac/04-22-2020COVIDC.xls")
## New names:
## * `` -> ...12
covidFIPSNum <- covidData[c("FIPS", "Confirmed")]
library(readxl)
cardioData <- read_excel("~/Desktop/0423/Age&Cardio/mapping_cardiac/Cardiovascular2014-.xlsx")
cardioFIPSNum <- cardioData[c("FIPS", "Mortality")]
total <- merge(covidFIPSNum,cardioFIPSNum,by="FIPS")
#visualise data
plot(as.numeric(total$Mortality), log(total$Confirmed), 
     xlab="Mortality of cardiovascular diseases per 10 000", ylab="Confirmed Covid-19 Cases", pch=16)

#correlation score
cor.test(x=as.numeric(total$Mortality),y=total$Confirmed,method="spearman")
